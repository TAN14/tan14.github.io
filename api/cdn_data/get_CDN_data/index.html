获取CDN数据(New)

TAN14 CDN 通过静态protobuf数据文件缓冲的形式为客户提供一个CDN数据的准实时观察效果

为了提供体积更小，解析更快，数据文件，我们使用了google的protobuf协议来构建数据。

数据文件位置

  1. 流量 proto文件：https://static.tan14.net/cdn_datas/stats.proto 数据文件： https://static.tan14.net/cdn_datas/CHANNEL_ID_OR_PN/bandwidth.protobuf
  2. 请求数和错误信息 proto文件：https://static.tan14.net/cdn_datas/cdn_info.proto 数据文件： https://static.tan14.net/cdn_datas/CHANNEL_ID_OR_PN/cdn_info.protobuf

URL变量说明

URL地址中大写的为变量, 其中:

  • PN 为您的产品ID
  • CHANNEL_ID 为您某个频道的ID

数据时间范围说明：

  • 流量文件存放距离当前时间半年内的数据
  • 请求数和错误信息文件存放距离当前时间一个月内的数据

注意事项：

app中展示的回源流量和回源请求是通过公式计算得出的，具体公式如下： 回源流量 = 上层回源流量 回源请求 = 上层回源请求

下面介绍一下数据中各项的含义:

  1. 流量proto文件（stats.proto）定义说明：

Text

syntax = "proto2";
package stats;

message Data {
    repeated uint64 flow = 1 [packed=true];
    #  flow结构定义如下 流量单位（字节）：
    #     （边缘流量, 上层流量, 回源流量） 
}

message Stats {
   repeated Data data = 1;
   repeated uint32 time = 2 [packed=true];
}
  1. 请求数和错误信息proto文件（cdn_info.proto）定义说明：

Text

syntax = "proto2";
package req_and_err;

message Req {
   repeated uint64 requests = 1 [packed=true];
   #  requests结构定义如下：
   #     (边缘请求数, 边缘动态请求数, 边缘回源请求数, 上层请求数, 上层回源请求数)
}

message E5xx {
   repeated uint32 err = 1 [packed=true];
   #  err结构定义如下：
   #     (边缘5xx, 边缘502, 边缘503, 边缘504, 上层5xx, 上层502, 上层503, 上层504)
}

message E4xx {
   repeated uint32 err = 1 [packed=true];
   #  err结构定义如下：
   #     (边缘4xx, 边缘403, 边缘404, 上层4xx, 上层403, 上层404)
}

message Info {
   repeated Req req = 1;
   repeated E5xx e5xx = 2;
   repeated E4xx e4xx = 3;
   repeated uint32 time = 4 [packed=true];
}
